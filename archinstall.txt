# Boot arch linux iso
ping -c 3 www.google.com #Check internet connectivity
efivar -l                        #If you get output, boot mode is UEFI
lsblk                                #Find your partition you want to install on
gdisk /dev/sdX                #Input x,z,y,y to wipe drive
cgdisk /dev/sdX                #New, [return], 1024MiB, ef00, EFI
                                #New, [return], [return], [return], root
                                #Write, yes
fdisk -l                        #Check partition layout
mkfs.fat -F32 /dev/sdX1        #Make EFI partition
mkfs.ext4 /dev/sdX2                #Make root partition
mount /dev/sdX2 /mnt
mkdir /mnt/boot
mount /dev/sdX1 /mnt/boot
pacman -Syy
pacman -S reflector
reflector -c Netherlands --sort country --sort delay --sort rate -p https -p http --save /etc/pacman.d/mirrorlist


pacstrap -i base base-devel neovim zsh wget linux-headers
intel-ucode networkmanager #Install intel-ucode only with intel


genfstab -U -p /mnt >> /mnt/etc/fstab


#Don’t install vi


arch-chroot /mnt        #Start chroot into new system
nvim /etc/locale.gen        #Uncomment your locale eg. nl_NL.UTF-8
locale-gen
echo LANG=nl_NL.UTF-8 > /etc/locale.conf
echo LC_ALL=nl_NL.UTF-8 > /etc/locale.conf
export LANG=nl_NL.UTF-8
ln -sf /usr/share/zoneinfo/Europe/Amsterdam /etc/localtime
hwclock --systohc --utc
systemctl enable fstrim.timer

________________

mount -t efivarfs efivarfs /sys/firmware/efi/efivars
bootctl install
nvim /boot/loader/entries/arch.conf #Enter this, with uuid of root from ‘sudo blkid -p /dev/sdX2’:
“
title   Arch Linux
linux   /vmlinuz-linux
initrd  /intel-ucode.img                #Only add this line with Intel
initrd  /initramfs-linux.img
options root=UUID=uuid rw
”
exit
umount -R /mnt
sudo reboot
